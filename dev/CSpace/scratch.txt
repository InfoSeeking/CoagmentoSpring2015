		function refresh2() {
			var maxChatID = document.getElementById('maxChatID');
			var maxChatValue = maxChatID.value;
			var url = "http://localhost/coagmento.com/CSpace/chatNew.php";
			req = new phpRequest(url);
			var response = req.execute();	
			if (response!=maxChatValue) {
				maxChatID.value = response;
				ajaxpage('sidebarChat.php','chat');
				var chatMessages = document.getElementById('chatMessages');
				chatMessages.scrollTop = chatMessages.scrollHeight;
			}
		}
